    # 根据当前时间获取这一周的时间范围

    TIP

    学习链接：https://www.cnblogs.com/blogs-8888/p/9585293.html

    # js获取一周的日期范围： getWeekType()这个方法如果不传参数 或者传入0，返回的是本周的日期范围，如果要下周的范围则传入1，上周的传入-1；注：这个是从周一开始算一周的开始，周日为结束。

    function getWeek() {
    this.nowTime = new Date();
    this.init = function() {
        this.dayInWeek = this.nowTime.getDay();
        this.dayInWeek == 0 && (this.dayInWeek = 7);
        this.thsiWeekFirstDay = this.nowTime.getTime() - (this.dayInWeek - 1) * 86400000;
        this.thisWeekLastDay = this.nowTime.getTime() + (7 - this.dayInWeek) * 86400000;
        return this;
    };
    this.getWeekType = function(type) {
        type = ~~type;
        var firstDay = this.thsiWeekFirstDay + type * 7 * 86400000;
        var lastDay = this.thisWeekLastDay + type * 7 * 86400000;
        return this.getWeekHtml(firstDay, lastDay);
    }
    this.formateDate = function(time) {
        var newTime = new Date(time)
        var year = newTime.getFullYear();
        var month = newTime.getMonth() + 1;
        var day = newTime.getDate();
        return year + "-" + (month >= 10 ? month : "0" + month) + "-" + (day >= 10 ? day : "0" + day);
    };
    this.getWeekHtml = function(f, l) {
        return this.formateDate(f) + "至" + this.formateDate(l);
    };
}

var getWeek = new getWeek();
var week = getWeek.init().getWeekType();
console.log(week);

    # 示例：

    <template>
    <div>
        <div>今天日期：{{$moment(new Date()).format("YYYY-MM-DD")}}</div>
        <div>这一周为：{{getWeek}}</div>
    </div>
</template>
<script>
export default {
    name:'week',
    computed:{
        getWeek(input) {
            if(input){
                let nowTime = new Date();
                let dayInWeek = nowTime.getDay(); //返回一周的某一天
                dayInWeek == 0 && (dayInWeek = 7);
                let thisWeekFirstDay = nowTime.getTime() - (dayInWeek - 1) * 86400000; //这周第一天
                let thisWeekLastDay = nowTime.getTime() + (7 - dayInWeek) * 86400000; //这周最后一天
                return this.$moment(thisWeekFirstDay).format("YYYY-MM-DD") + " 至 " + this.$moment(thisWeekLastDay).format("YYYY-MM-DD")
            }
        }
    }
}
</script>

    Vue.filter("writeRange", function(value, formatString) {
    if(value){
        formatString = formatString || "YYYY-MM-DD";
        let date = moment(value).format(formatString);
        let nowTime = new Date(date);
        let dayInWeek = nowTime.getDay(); //返回一周的某一天
        dayInWeek == 0 && (dayInWeek = 7);
        let thisWeekFirstDay = nowTime.getTime() - (dayInWeek - 1) * 86400000; //这周第一天
        let thisWeekLastDay = nowTime.getTime() + (7 - dayInWeek) * 86400000; //这周最后一天
        return moment(thisWeekFirstDay).format(formatString) + " 至 " + moment(thisWeekLastDay).format(formatString)
    }else{
        return ""
    }
});
